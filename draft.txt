// Template definitions

// Create Template Request
enum ItemAcceptedTypes { 
    number = "number",
    string = "string",
    boolean = "boolean",
    date = "date",
    enum = "enum",
}

interface ItemType {
    type: ItemAcceptedTypes,
    default?: T<ItemAcceptedTypes>,
    rules?: string[],
    values?: string[], //obrigatorio quando o type for enum
}

interface TemplateItem {
    name: string,
    type: ItemType,
    required: boolean,
    rules?: string[], // regra para o campo
    description?: string,
}

interface Template {
    name: string,
    description?: string,
    items: TemplateItem[],
    rules?: string[], // regra para o template
}
